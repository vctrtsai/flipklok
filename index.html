<script>

let lastMinute = null;


function updateClock() {

  const now = new Date();

  const brisbane = new Date(

    now.toLocaleString('en-US', { timeZone: 'Australia/Brisbane' })

  );


  let hours = brisbane.getHours();

  const minutes = brisbane.getMinutes();

  const ampm = hours >= 12 ? 'PM' : 'AM';


  hours = hours % 12;

  hours = hours ? hours : 12;


  if (minutes !== lastMinute) {

    document.getElementById('hourCard').classList.remove('flip');

    document.getElementById('minuteCard').classList.remove('flip');

    void document.body.offsetWidth;

    document.getElementById('hourCard').classList.add('flip');

    document.getElementById('minuteCard').classList.add('flip');

    lastMinute = minutes;

  }


  document.getElementById('hour').textContent = hours;

  document.getElementById('minute').textContent = String(minutes).padStart(2, '0');

  document.getElementById('ampm').textContent = ampm;

}


updateClock();

setInterval(updateClock, 1000);
